 20 /**
 21  * This class represents the keyboard and handle key states
 22  *
 23  * @author Mounier Florian
 24  * @constructor
 25  *
 26  */
 27 var Keyboard = function () {
 28 };
 29 
 30 /**
 31  * This method handle a key down
 32  */
 33 Keyboard.prototype.down = function(event) {
 34     $.each(Jspung._.pung.players, function (i, player) {
 35 	       $.each(player.controls, function (k, o) {
 36 			  if(event.keyCode == o.keyCode) {
 37 			      o.down = true;
 38 			  }
 39 		      });
 40 	   });
 41 };
 42 
 43 /**
 44  * This method handle a key up
 45  */
 46 Keyboard.prototype.up = function(event) {
 47     $.each(Jspung._.pung.players, function (i, player) {
 48 	       $.each(player.controls, function (k, o) {
 49 			  if(event.keyCode == o.keyCode) {
 50 			      o.down = false;
 51 			  }
 52 		      });
 53 	   });
 54 };
 55 
 56 Keyboard.prototype.toString = function ()  {
 57     return "Keyboard";
 58 };
