 20 /**
 21  * This class represents a geometric location
 22  *
 23  * @author Mounier Florian
 24  * @constructor
 25  * @param x Initial abscissa
 26  * @param y Initial ordinate
 27  * @param speed Location speed
 28  *
 29  */
 30 var Location = function (x, y, speed) {
 31     this.x = x;
 32     this.y = y;
 33     this.speed = speed;
 34 };
 35 
 36 /**
 37  * This method computes the location change in dt ms depending on the speed and the acceleration
 38  * @param dt Time delta to compute move
 39  */
 40 Location.prototype.move = function(dt) {
 41     this.speed.accelerate(dt);
 42     this.x += (this.speed.x * dt) / 1000;
 43     this.y += (this.speed.y * dt) / 1000;
 44 };
 45 
 46 /**
 47  * This method is similar to move but move backwards
 48  * @param dt Time delta to compute move
 49  */
 50 Location.prototype.moveBackwards = function(dt) {
 51     this.speed.accelerate(dt);
 52     this.x -= (this.speed.x * dt) / 1000;
 53     this.y -= (this.speed.y * dt) / 1000;
 54 };
 55 /**
 56  * This method computes the distance between this location and another
 57  * @param other The other location
 58  */
 59 Location.prototype.distanceTo = function(other) {
 60     return Math.sqrt(Math.pow(this.x - other.x, 2) + Math.pow(this.y - other.y, 2));
 61 };
 62 
 63 Location.prototype.toString = function ()  {
 64     return "L{x: " + this.x + ", y:" + this.y + "} " + this.speed;
 65 };
