 20 /**

 22  * @constructor
 23  * @author Mounier Florian

 26     this.canvas = $('#canvas')[0];
 27     this.c = this.canvas.getContext('2d');
 28     this.screen = new Screen(this.canvas);
 29     this.keyboard = new Keyboard();
 30     this.frames = 0;

 74 /**
 75  * This method starts the fps counter
 76  */
 77 Jspung.prototype.startFps = function () {
 78     setInterval(function () {Jspung._.fps();}, 1000);
 79 };
 80 
 81 /**
 82  * This method is periodically called to display the fps rate
 83  */
 84 
 85 Jspung.prototype.fps = function () {
 86     document.title = this.frames + " fps";
 87     this.frames = 0;
 88 };
 89 
 90 /**
 91  * This method is periodically called to call the animate function of the game
 92  * and to check the state of the game
 93  */
 94 Jspung.prototype.animate = function () {
 95     this.frames++;
 96     var gameState = this.pung.animate();
 97     if(!gameState) {
 98 	// The players are all dead
 99 	if(gameState == null) {
100 	    this.makeLevel();
101 	} else {
102 	    this.level++;
103 	    this.makeLevel();
104 	}
105     }
106     setTimeout(function () { Jspung._.animate(); }, 8); // ~120Hz
107 };
108 
109 /**
110  * This function create a new level according to the value of level
111  */
112 Jspung.prototype.makeLevel = function () {
113     this.pung = new Pung(this.c);
114     var levelBalls = Jspung.ballsByLevel[this.level - 1];
115     var _this = this;
116     $.each(levelBalls, function (i, ball) {
117 	       _this.pung.addBall(new Ball(ball.x, ball.xspeed, ball.life));
118 	   });
119     this.pung.addPlayer(new Player(this.screen.w / 3, 1, this.leftPlayerSet));
120     this.pung.addPlayer(new Player(this.screen.w * 2 / 3, 2, this.rightPlayerSet));
121 
122     // Init canvas
123     this.c.save();
124     this.c.fillStyle = "rgb(34, 34, 34)";
125     this.c.fillRect(0, 0, this.screen.w, this.screen.h);
126     this.c.restore();
127 };
