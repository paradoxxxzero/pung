 20 /**
 21  * This class represents the screen
 22  *
 23  * @author Mounier Florian
 24  * @constructor
 25  * @param canvas The canvas element
 26  *
 27  */
 28 var Screen = function (canvas) {
 29     this.canvas = canvas;
 30     this.w = canvas.width = $("#canvas").width();  //window.innerWidth;
 31     this.h = canvas.height = $("#canvas").height(); //window.innerHeight;
 32 };
 33 
 34 /**
 35  * This method refresh the canvas on resize
 36  * Unused since we use fixed size now
 37  */
 38 Screen.prototype.resize = function() {
 39     this.w = this.canvas.width = window.innerWidth;
 40     this.h = this.canvas.height = window.innerHeight;
 41     // Background initialisation
 42     var context = this.canvas.getContext('2d');
 43     //TODO clip
 44     context.save();
 45     context.fillStyle = "rgba(34, 34, 34, 1)";
 46     context.fillRect(0, 0, this.w, this.h);
 47     context.restore();
 48 };
 49 
 50 Screen.prototype.toString = function ()  {
 51     return "Scr{w: " + this.w + ", h:" + this.h + "} ";
 52 };
