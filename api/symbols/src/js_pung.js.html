 20 /**
 21  * This class represents a pung game
 22  *
 23  * @author Mounier Florian
 24  * @constructor
 25  * @param context The 2d canvas context
 26  *
 27  */
 28 var Pung = function (context) {
 29     this.balls = new Array();
 30     this.players = new Array();
 31     this.bullets = new Array();
 32     this.grapnels = new Array();
 33     this.objects = [this.balls, this.bullets, this.grapnels, this.players];
 34     this.context = context;
 35     this.time = new Date().getTime();
 36 };
 37 
 38 /**
 39  * This method adds a ball to the game
 40  * @param ball The Ball to add
 41  */
 42 Pung.prototype.addBall = function(ball) {
 43     this.balls.push(ball);
 44 };
 45 
 46 /**
 47  * This methods erases a ball from the game
 48  * @param ball The Ball to remove
 49  */
 50 Pung.prototype.rmBall = function(ball) {
 51     this.balls.splice(this.balls.indexOf(ball), 1);
 52 };
 53 
 54 /**
 55  * This method adds a bullet to the game
 56  * @param bullet The Bullet to add
 57  */
 58 Pung.prototype.addBullet = function(bullet) {
 59     this.bullets.push(bullet);
 60 };
 61 
 62 /**
 63  * This methods erases a bullet from the game
 64  * @param bullet The Bullet to remove
 65  */
 66 Pung.prototype.rmBullet = function(bullet) {
 67     this.bullets.splice(this.bullets.indexOf(bullet), 1);
 68 };
 69 
 70 /**
 71  * This method adds a grapnel to the game
 72  * @param grapnel The Grapnel to add
 73  */
 74 Pung.prototype.addGrapnel = function(grapnel) {
 75     this.grapnels.push(grapnel);
 76 };
 77 
 78 /**
 79  * This methods erases a bullet from the game
 80  * @param bullet The Bullet to remove
 81  */
 82 Pung.prototype.rmGrapnel = function(grapnel) {
 83     this.grapnels.splice(this.bullets.indexOf(grapnel), 1);
 84 };

 86 /**
 87  * This method adds a player to the game
 88  * @param player The Player to add
 89  */
 90 Pung.prototype.addPlayer = function(player) {
 91     this.players.push(player);
 92 };
 93 
 94 /**
 95  * This methods suppress a player from the game
 96  * @param player The Player to remove
 97  */
 98 Pung.prototype.rmPlayer = function(player) {
 99     this.players.splice(this.players.indexof(player), 1);
100 };
101 
102 /**
103  * This method makes the game
104  *
105  * @param ball The Ball to add
106  * @return True if the game goes on
107  *         False if the players won this level
108  *         Null if the level is lost
109  */
110 Pung.prototype.animate = function() {
111     var _this = this;
112     var c = this.context;
113 
114     // Handle player action (bullets, grapnels...)
115     $.each(this.players, function(i, player) {
116 	       if(player.controls.bullet.down) {
117 		   _this.addBullet(new Bullet(player));
118 		   player.controls.bullet.down = false;
119 	       }
120 	       if(player.controls.grapnel.down && player.shotsLeft > 0) {
121 		   player.shotsLeft--;
122 		   _this.addGrapnel(new Grapnel(player));
123 		   player.controls.grapnel.down = false;
124 	       }
125 	   });
126 
127    // Clip all objects
128     c.save();
129     c.fillStyle = "rgb(34, 34, 34)";
130     $.each(this.objects, function(i, os) {
131 	       $.each(os, function(j, o) {
132 			  o.clip(c);
133 		      });
134 	   });
135     c.restore();
136 
137     // Move all objects
138     // This array stores an object containing the object to destroy and his array
139     var toBeDestroyed = [];
140     var dt = Math.min(new Date().getTime() - this.time, 50);
141     $.each(this.objects, function(i, os) {
142 	       $.each(os, function(j, o) {
143 			  if(o.move(dt)) {
144 			      toBeDestroyed.push({os: os, o: o});
145 			  }
146 		      });
147 	   });
148 
149     // Handling ball collision with bullets or grapnels
150     $.each(this.balls, function(i, ball) {
151 	       $.each(_this.bullets, function(j, bullet) {
152 			  if(bullet.isCollidingWith(ball)) {
153 			      bullet.player.increaseScore(50);
154 			      toBeDestroyed.push({os: _this.balls, o: ball});
155 			      toBeDestroyed.push({os: _this.bullets, o: bullet});
156 			      var forks = ball.fork();
157 			      if(forks != null) {
158 				  _this.balls.push(forks.left);
159 				  _this.balls.push(forks.right);
160 			      }
161 			  }
162 		      });
163 	       $.each(_this.grapnels, function(j, grapnel) {
164 			  if(grapnel.isCollidingWith(ball)) {
165 			      grapnel.player.increaseScore(100);
166 			      grapnel.player.shotsLeft++;
167 			      toBeDestroyed.push({os: _this.balls, o: ball});
168 			      toBeDestroyed.push({os: _this.grapnels, o: grapnel});
169 
170 			      var forks = ball.fork();
171 			      if(forks != null) {
172 				  _this.balls.push(forks.left);
173 				  _this.balls.push(forks.right);
174 			      }
175 			  }
176 		      });
177 	       $.each(_this.players, function(j, player) {
178 			  if(player.isCollidingWith(ball)) {
179 			      toBeDestroyed.push({os: _this.players, o: player});
180 			  }
181 		      });
182 	   });
183 
184     // Remove destroyed objects
185     $.each(toBeDestroyed, function(i, oz) {
186 	       oz.os.splice(oz.os.indexOf(oz.o), 1);
187 	   });
188 
189     // Render all objects
190     $.each(this.objects, function(i, os) {
191 	       c.save();
192 	       $.each(os, function(j, o) {
193 			  o.render(c);
194 		      });
195 	       c.restore();
196 	   });
197 
198     // Test end of level
199     if(this.balls.length == 0) {
200 	return false;
201     }
202     if(this.players.length == 0) {
203 	return null;
204     }
205     this.time = new Date().getTime();
206     return true;
207 };
208 
209 
210 Pung.prototype.toString = function ()  {
211     return "Pung game";
212 };
