 20 /**
 21  * This class represents a speed for a location
 22  *
 23  * @author Mounier Florian
 24  * @constructor
 25  * @param x Initial speed abscissa
 26  * @param y Initial speed ordinate
 27  * @param acceleration Location acceleration
 28  *
 29  */
 30 var Speed = function (x, y, acceleration) {
 31     this.x = x;
 32     this.y = y;
 33     this.acceleration = acceleration;
 34 };
 35 
 36 /**
 37  * This method compute the speed change in dt ms depending on the acceleration
 38  * @param dt Time delta to compute acceleration
 39  */
 40 Speed.prototype.accelerate = function (dt) {
 41     this.x += (this.acceleration.x * dt) / 1000;
 42     this.y += (this.acceleration.y * dt) / 1000;
 43 };
 44 
 45 Speed.prototype.toString = function ()  {
 46     return "S{x: " + this.x + ", y:" + this.y + "} " + this.acceleration;
 47 };
