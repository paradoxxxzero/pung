 20 /**
 21  * This class represents a bullet
 22  *
 23  * @author Mounier Florian
 24  * @constructor
 25  * @param player The player who shot this bullet
 26  *
 27  */
 28 
 29 var Bullet = function (player) {
 30     this.location = new Location(
 31 	player.location.x, Jspung._.screen.h - player.shape.h,
 32 	new Speed(
 33 	    0, -2000,
 34 	    new Acceleration(0, 0)
 35 	));
 36     this.shape = new Shape(2, 10);
 37     this.color = Bullet.colors[player.index];
 38     this.player = player;
 39 };
 40 
 41 /**
 42  * Static bullet colors
 43  */
 44 Bullet.colors = new Array();
 45 $(document).ready(function () {
 46 		      for (var p = 0 ; p <= Player.maxPlayers ; p++) {
 47 			  Bullet.colors[p] = $("div.bullet-" + p).css("color");
 48 		      }
 49 		  });
 50 
 51 /**
 52  * This method moves the bullet
 53  * @param dt Time delta to compute move
 54  * @return true If the bullet must be destroyed
 55  */
 56 Bullet.prototype.move = function(dt) {
 57     this.location.move(dt);
 58     return this.location.y < 0;
 59 };
 60 
 61 /**
 62  * This method renders the bullet
 63  * @param c The canvas context
 64  */
 65 Bullet.prototype.render = function(c) {
 66     c.shadowBlur = 5;
 67     c.shadowColor = this.color;
 68     c.fillStyle = this.color;
 69     c.fillRect(Math.round(this.location.x - this.shape.w / 2),
 70 	       Math.round(this.location.y),
 71 	       this.shape.w,
 72 	       this.shape.h);
 73 };
 74 
 75 /**
 76  * This method clip what was previously rendered
 77  * @param c The canvas context
 78  */
 79 Bullet.prototype.clip = function(c) {
 80     var margin = 5;
 81     c.fillRect(Math.round(this.location.x - this.shape.w / 2 - margin),
 82 	       Math.round(this.location.y - margin),
 83 	       this.shape.w + 2 * margin,
 84 	       this.shape.h + 2 * margin);
 85 };
 86 
 87 /**
 88  * This method tests if the bullet is colliding with the ball
 89  * @param ball The ball to test the collision with
 90  */
 91 Bullet.prototype.isCollidingWith = function(ball) {
 92     return this.location.distanceTo(ball.location) < ball.radius;
 93 };
 94 
 95 
 96 Bullet.prototype.toString = function ()  {
 97     return "Bullet " + this.location;
 98 };
