 20 /**
 21  * This class represents a simple ball
 22  * This constructor initialize y speed to 0 and acceleration to standard gravity acceleration
 23  *
 24  * @author Mounier Florian
 25  * @constructor
 26  * @param x The initial ball abscissa
 27  * @param xspeed The initial ball horizontal speed
 28  * @param life Life of the ball defines how many hit remains before the ball dies
 29  *
 30  */
 31 
 32 var Ball = function (x, xspeed, life) {
 33     this.life = life;
 34     this.location = new Location(
 35 	x, JSPung._.screen.h * Ball.heights[this.life],
 36 	new Speed(
 37 	    xspeed, 0,
 38 	    new Acceleration(0, 980)
 39 	));
 40     this.radius = Ball.sizes[life];
 41     this.color = Ball.colors[life];
 42 };
 43 
 44 /**
 45  * Static sizes for balls life
 46  */
 47 Ball.sizes = [6.25, 12.5, 25, 50, 100];
 48 
 49 /**
 50  * Static max heights for balls
 51  */
 52 Ball.heights = [.75, .625, .5, .25, .125];
 53 
 54 /**
 55  * Static max life for balls
 56  */
 57 Ball.maxLife = 4;
 58 
 59 /**
 60  * Static balls colors
 61  */
 62 Ball.colors = new Array();
 63 $(document).ready(function () {
 64 		      for (var l = 0 ; l <= Ball.maxLife ; l++) {
 65 			  Ball.colors[l] = $("div.ball-" + l).css("color");
 66 		      }
 67 		  });
 68 /**
 69  * This method moves the ball
 70  * @param dt Time delta to compute move
 71  * @return true If the ball must be destroyed
 72  */
 73 Ball.prototype.move = function(dt) {
 74     this.location.move(dt);
 75     if(this.location.x + this.radius > JSPung._.screen.w) {
 76 	this.location.x = JSPung._.screen.w - this.radius;
 77 	this.location.speed.x *= -1;
 78     } else if(this.location.x - this.radius < 0) {
 79 	this.location.x = this.radius;
 80 	this.location.speed.x *= -1;
 81     }
 82     if(this.location.y + this.radius > JSPung._.screen.h) {
 83 	this.location.y = JSPung._.screen.h - this.radius;
 84 	this.location.speed.y *= -1;
 85     }
 86     return false;
 87 };
 88 
 89 /**
 90  * This method renders the ball
 91  * @param c The canvas context
 92  */
 93 Ball.prototype.render = function(c) {
 94     c.beginPath();
 95     c.shadowBlur = 10;
 96     c.shadowColor = this.color;
 97     c.fillStyle = this.color;
 98     c.arc(this.location.x, this.location.y, this.radius, 0, 2 * Math.PI, false);
 99     c.fill();
100 };
101 
102 /**
103  * This method forks a ball after a collision
104  * @return an object with the two forked balls one .left and one .right or null if this ball has no more life
105  */
106 Ball.prototype.fork = function() {
107     if(this.life == 0) return null;
108     var left = new Ball(this.location.x, -this.location.speed.x, this.life - 1, this.color);
109     var right = new Ball(this.location.x, this.location.speed.x, this.life - 1, this.color);
110     return {left: left, right: right};
111 };
112 
113 Ball.prototype.toString = function ()  {
114     return "Ball R:" + this.radius + " V:" + this.life + " " + this.location;
115 };
