 20 /**
 21  * This class represents a player
 22  * This constructor initializes the player with a default shape
 23  * @author Mounier Florian
 24  * @constructor
 25  * @param x The initial player abscissa
 26  * @param index The player index (1 for p1, 2 for p2, ...)
 27  * @param controls The keyboard keyCodes to control this player
 28  *
 29  */
 30 var Player = function (x, index, controls) {
 31     this.shape = new Shape(20,40);
 32     this.location = new Location(
 33 	x, JSPung._.screen.h - this.shape.h,
 34 	new Speed(
 35 	    500, 0,
 36 	    new Acceleration(0, 0)
 37 	)
 38     );
 39     this.index = index;
 40     this.controls = controls;
 41     this.color = Player.colors[index];
 42 };
 43 
 44 /**
 45  * Static max player number (for colors)
 46  */
 47 Player.maxPlayers = 2;
 48 
 49 /**
 50  * Static player colors
 51  */
 52 Player.colors = new Array();
 53 $(document).ready(function () {
 54 		      for (var p = 0 ; p <= Player.maxPlayers ; p++) {
 55 			  Player.colors[p] = $("div.player-" + p).css("color");
 56 		      }
 57 });
 58 
 59 /**
 60  * This method moves the player
 61  * @param dt Time delta to compute move
 62  * @return true If the player must be destroyed
 63  */
 64 Player.prototype.move = function(dt) {
 65     if(this.controls.right.down && this.location.x + this.shape.w / 2 < JSPung._.screen.w) {
 66 	this.location.move(dt);
 67     }
 68     if(this.controls.left.down && this.location.x - this.shape.w / 2 > 0) {
 69 	this.location.moveBackwards(dt);
 70     }
 71     return false;
 72 };
 73 /**
 74  * This method renders the player
 75  * @param c The canvas context
 76  */
 77 Player.prototype.render = function(c) {
 78     c.shadowBlur = 5;
 79     c.shadowColor = this.color;
 80     c.fillStyle = this.color;
 81     c.fillRect(this.location.x - this.shape.w / 2, this.location.y, this.shape.w, this.shape.h);
 82 };
 83 
 84 /**
 85  * This method tests if the player is colliding with the ball
 86  * @param ball The ball to test the collision with
 87  */
 88 Player.prototype.isCollidingWith = function(ball) {
 89     var originalX = this.location.x;
 90     this.location.x -= this.shape.w / 2;
 91     // Approx collision by only computing the two top corners of the player :
 92     var colliding = this.location.distanceTo(ball.location) < ball.radius;
 93     this.location.x += this.shape.w;
 94     colliding = colliding || this.location.distanceTo(ball.location) < ball.radius;
 95     this.location.x = originalX;
 96     return colliding;
 97 };
 98 
 99 Player.prototype.toString = function ()  {
100     return "Player " + this.index + " " + this.location;
101 };
