 21 /**
 22  * This class represents a grapnel
 23  *
 24  * @author Mounier Florian 
 25  * @constructor
 26  * @param location Initial location
 27  * @param shape The grapnel shape (h is not used)
 28  *
 29  */
 30 
 31 var Grapnel = function (location, shape) {
 32     this.location = location;
 33     this.shape = shape;
 34 
 35 };
 36 
 37 /**
 38  * This method moves the grapnel
 39  * @param dt Time delta to compute move
 40  * @return true If the grapnel must be destroyed
 41  */
 42 Grapnel.prototype.move = function(dt) {
 43     this.location.move(dt);
 44     return this.location.y > 0;
 45 };
 46 
 47 /**
 48  * This method renders the bullet
 49  * @param c The canvas context
 50  */
 51 Grapnel.prototype.render = function(c) {
 52     c.shadowBlur = 5;
 53     c.shadowColor = _colors.grapnel;
 54     c.fillStyle = _colors.grapnel;
 55     c.fillRect(this.location.x - this.shape.w / 2, this.location.y, this.shape.w, _scr.h - this.location.y);
 56 };
