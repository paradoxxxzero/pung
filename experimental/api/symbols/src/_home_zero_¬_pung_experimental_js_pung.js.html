 21 /**
 22  * This class represents a pung game
 23  *
 24  * @author Mounier Florian 
 25  * @constructor
 26  *
 27  */
 28 var Pung = function (context) {
 29     this.balls = new Array();
 30     this.players = new Array();
 31     this.context = context;
 32     this.time = new Date().getTime();
 33 };
 34 
 35 /**
 36  * This method adds a ball to the game
 37  * @param ball The Ball to add
 38  */
 39 Pung.prototype.addBall = function(ball) {
 40     this.balls.push(ball);
 41 };
 42 
 43 /**
 44  * This methods erases a ball from the game
 45  * @param ball The Ball to remove
 46  */
 47 Pung.prototype.rmBall = function(ball) {
 48     this.balls.splice(this.balls.indexOf(ball), 1);
 49 };
 50 
 51 /**
 52  * This utility method creates a new ball with default values
 53  * @param x The initial ball abscissa
 54  * @param xspeed The initial ball horizontal speed
 55  * @param life The initial ball life
 56  */
 57 Pung.prototype.makeBall = function(x, xspeed, life) {
 58     return new Ball(
 59 	new Location(
 60 	    x, _scr.h * _heights[life - 1],
 61 	    new Speed(
 62 		xspeed, 0,
 63 		new Acceleration(0, 980)
 64 	    )
 65 	), life);
 66 };
 67 
 68 /**
 69  * This method adds a player to the game
 70  * @param player The Player to add
 71  */
 72 Pung.prototype.addPlayer = function(player) {
 73     this.players.push(player);
 74 };
 75 
 76 /**
 77  * This methods suppress a player from the game
 78  * @param player The Player to remove
 79  */
 80 Pung.prototype.rmPlayer = function(player) {
 81     this.players.splice(this.players.indexof(player), 1);
 82 };
 83 
 84 /**
 85  * This utility method creates a new player with default values
 86  * @param x The initial player abscissa
 87  */
 88 Pung.prototype.makePlayer = function(x) {
 89     return new Player(
 90 	new Location(
 91 	    x, _scr.h,
 92 	    new Speed(
 93 		500, 0,
 94 		new Acceleration(0, 0)
 95 	    )
 96 	), new Shape(20,40));
 97 };
 98 
 99 /**
100  * This method makes the game
101  * @param ball The Ball to add
102  */
103 Pung.prototype.animate = function() {
104     var c = this.context;
105     var dt = Math.min(new Date().getTime() - this.time, 50);
106 
107     $.each(this.balls.concat(this.players), function(i, o) {
108 	       o.move(dt);
109 	   });
110 
111     $.each(this.balls.concat(this.players), function(i, o) {
112 	       o.render(c);
113 	   });
114 
115     this.time = new Date().getTime();
116 };
